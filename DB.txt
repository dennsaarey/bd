CREATE TABLE ASESOR(
IDASESOR VARCHAR2(11) NOT NULL,
CONTRASENIA VARCHAR2(20) NOT NULL,
NOMBRE VARCHAR2(30) NOT NULL,
APELLPATERNO VARCHAR2(30) NOT NULL,
APELLMATERNO VARCHAR2(30) NULL,
CUBICULO VARCHAR2(30) NOT NULL,
CORREOELECTRONICO VARCHAR2(30) NULL,
HORARIODEASESORIA VARCHAR2(15) NOT NULL,
CONSTRAINT PK_IDASESOR PRIMARY KEY(IDASESOR)
);

CREATE TABLE CANDIDATO(
IDCANDIDATO VARCHAR2(11) NOT NULL,
IDASESOR VARCHAR2(11) NOT NULL,
NOMBRE VARCHAR2(30) NOT NULL,
APELLPATERNO VARCHAR2(30) NOT NULL,
APELLMATERNO VARCHAR2(30) NULL,
FOTO VARCHAR2(30) NOT NULL,
CORREOELECTRONICO VARCHAR2(30) NULL,
TELEFONOCELULAR NUMBER(13) NULL,
CARRERA VARCHAR2(30) NOT NULL,
GENERACION VARCHAR2(20) NOT NULL,
TEMADETESIS VARCHAR2(30) NOT NULL,
DIRECTORDETESIS VARCHAR2(30) NOT NULL,
TRABAJO VARCHAR2(30) NOT NULL,
LUGARDETRABAJO VARCHAR2(30) NULL,
HORARIODETRABAJO VARCHAR2(15) NULL,
ADMITIDO VARCHAR2(2) NOT NULL,
COMPROMISO VARCHAR2(500) NULL,
EXPOSICIONDEMOTIVOS VARCHAR2(500) NULL,
CONSTRAINT PK_IDCANDIDATO PRIMARY KEY(IDCANDIDATO),
FOREIGN KEY(IDASESOR) REFERENCES ASESOR(IDASESOR)
);


CREATE TABLE ADMINISTRADOR(
IDADMINISTRADOR VARCHAR2(11) NOT NULL,
NOMBRE VARCHAR2(30) NOT NULL,
APELLPATERNO VARCHAR2(30) NOT NULL,
APELLMATERNO VARCHAR2(30) NULL,
CONTRASENIA VARCHAR2(20) NOT NULL
);
